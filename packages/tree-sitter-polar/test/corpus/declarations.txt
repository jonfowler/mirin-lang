cmp multAdd
  { #clk: Clock, rstn: Reset @clk = high, c: uint[8] @clk = 0, }
  ( a: uint[8] @clk, b: uint[8] @clk )
  -> uint[8] @clk
  {
    let mult = a * b;
    let mult = mult[8:0];
    let mult = mult.reg{rstn}();
    let add = mult + c;
    return add;
  }

struct Bus(A: Type) = bus {
  valid: bool,
  data: A,
}

port DF
  { #clk: Clock }
  ( A: Type )
  = df {
    in ready: bool @clk,
    out valid: bool @clk,
    out data: A @clk,
  }

cmp connectDf
  { #clk: Clock }
  ( upstream: DF{clk}(uint[8]), out downstream: DF{clk}(uint[8]) )
  {
    downstream.valid = upstream.valid;
    downstream.data = upstream.data;
    upstream.ready = downstream.ready;
  }
