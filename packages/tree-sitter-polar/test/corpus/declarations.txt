  { #clk: Clock, rstn: Reset @clk = high, c: uint(8) @clk = 0, }
  ( a: uint(8) @clk, b: uint(8) @clk )
  -> uint(8) @clk
  {
    let mult = a * b;
    let mult = mult.reg{rstn}();
    let add = mult + c;
    return add;
  }
